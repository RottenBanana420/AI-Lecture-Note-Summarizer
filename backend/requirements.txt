# Core FastAPI Framework
fastapi  # Modern, fast web framework for building APIs with Python 3.7+

# ASGI Server
uvicorn  # Lightning-fast ASGI server implementation for running FastAPI applications

# Database ORM and Driver
sqlalchemy  # SQL toolkit and Object-Relational Mapping (ORM) library
psycopg2-binary  # PostgreSQL database adapter for Python (binary distribution)

# Database Migrations
alembic  # Lightweight database migration tool for SQLAlchemy

# Environment Variables
python-dotenv  # Read key-value pairs from .env file and set them as environment variables
pydantic-settings  # Settings management using Pydantic for type-safe configuration

# Vector Database Support
pgvector  # PostgreSQL extension for vector similarity search with embeddings

# Testing Framework
pytest  # Modern testing framework with powerful features and plugins
pytest-asyncio  # Pytest plugin for testing asyncio code
pytest-cov  # Coverage plugin for pytest using coverage.py
pytest-xdist  # Pytest plugin for distributed/parallel test execution
httpx  # HTTP client for testing FastAPI applications (used by TestClient)
